create database Empresa;



use Empresa;





create table empresa(

empresaNome char(100) not null,

empresaCNPJ char(100) not null,

empresaRazaoSocial char(100) not null,

empresaConjunto int(100) not null,

empresaAbertura varchar(100) not null,

empresaEncerramento varchar(100) not null,

empresaTemperaturaMin int(100) not null,

empresaTemperaturaMax int(100) not null,

horarioArLigado varchar(100) not null,

horarioArDesligado varchar(100) not null,

primary key(empresaNome),

unique(empresaCNPJ),

unique(empresaRazaoSocial),

unique(empresaConjunto)
)
;




create table usuario(
usuarioNome varchar(10) not null,

usuarioRG varchar(30) not null,

usuarioCPF varchar(30) not null,

usuarioSexo varchar(50) not null,

usuarioConta varchar(10) not null,

usuarioSenha varchar(10) not null,

usuarioHorarioEntrada varchar(100) not null,

usuarioHorarioSaida varchar(100) not null,

usuarioTipo varchar(100) not null,

primary key(usuarioNome),

unique(usuarioRG),

unique(usuarioCPF),

unique(usuarioConta)
)
;
